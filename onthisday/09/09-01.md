<!-- Script for calculating number of years ago -->
<script>
var dob = '19740901';
var year = Number(dob.substr(0, 4));
var month = Number(dob.substr(4, 2)) - 1;
var day = Number(dob.substr(6, 2));
var today = new Date();
var age1 = today.getFullYear() - year;
